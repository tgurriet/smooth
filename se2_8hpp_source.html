193 SE2(const Eigen::Transform<Scalar, 2, Eigen::Isometry> & t)
194 {
195 Eigen::Matrix2<Scalar> rotmat = t.rotation();
196 coeffs().x() = t.translation().x();
197 coeffs().y() = t.translation().y();
198 coeffs().z() = rotmat(1, 0); // sin(angle)
199 coeffs().w() = rotmat(0, 0); // cos(angle)
