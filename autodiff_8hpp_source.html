43template<std::size_t K = 1>
44 requires(K >= 1 && K <= 2)
45auto dr_autodiff(auto && f, auto && x)
46{
47 using Result = decltype(std::apply(f, x));
48 using Scalar = ::smooth::Scalar<Result>;
49 using Eigen::Matrix;
50
51 static_assert(Manifold<Result>, "f(x) is not a Manifold");
52
53 using AdScalar = autodiff::HigherOrderDual<K, Scalar>;
54
55 Result F = std::apply(f, x);
56
57 static constexpr auto Nx = wrt_Dof<decltype(x)>();
58 static constexpr auto Ny = Dof<Result>;
59 const Eigen::Index nx = std::apply([](auto &&... args) { return (dof(args) + ...); }, x);
60 const Eigen::Index ny = dof(F);
61
62 // cast F and x to ad types
63 const auto x_ad = wrt_cast<AdScalar>(x);
64 CastT<AdScalar, Result> F_ad = cast<AdScalar>(F);
65
66 if constexpr (K == 1) {
67 // zero-valued tangent element
68 Matrix<AdScalar, Nx, 1> a_ad = Matrix<AdScalar, Nx, 1>::Zero(nx);
69
70 // function to differentiate
71 const auto f_ad = [&](Matrix<AdScalar, Nx, 1> & var) -> Matrix<AdScalar, Ny, 1> {
72 return rminus<CastT<AdScalar, Result>>(std::apply(f, wrt_rplus(x_ad, var)), F_ad);
73 };
74
75 Matrix<Scalar, Ny, Nx> J(ny, nx);
76
77 J = autodiff::jacobian(f_ad, autodiff::wrt(a_ad), autodiff::at(a_ad));
78 return std::make_pair(std::move(F), std::move(J));
79 } else if constexpr (K == 2) {
80 // function to differentiate
81 const auto f_ad =
82 [&f, &x_ad, &F_ad](Matrix<AdScalar, Nx, 1> & var1, Matrix<AdScalar, Nx, 1> & var2) -> Matrix<AdScalar, Ny, 1> {
83 return rminus(std::apply(f, wrt_rplus(wrt_rplus(x_ad, var1), var2)), F_ad);
84 };
85
86 Matrix<Scalar, Ny, Nx> J(ny, nx);
87 Matrix<Scalar, Nx, std::min(Nx, Ny) == -1 ? -1 : Nx * Ny> H(nx, nx * ny);
88
89 // zero-valued tangent elements
90 Matrix<AdScalar, Nx, 1> a_ad1 = Matrix<AdScalar, Nx, 1>::Zero(nx);
91 Matrix<AdScalar, Nx, 1> a_ad2 = Matrix<AdScalar, Nx, 1>::Zero(nx);
92
93 const auto a_wrt1 = autodiff::wrt(a_ad1);
94 const auto a_wrt2 = autodiff::wrt(a_ad2);
95
96 autodiff::detail::ForEachWrtVar(
97 a_wrt1, [&](auto && i, auto && xi) constexpr {
98 autodiff::detail::ForEachWrtVar(
99 a_wrt2, [&](auto && j, auto && xj) constexpr {
100 const auto u = autodiff::detail::eval(f_ad, autodiff::at(a_ad1, a_ad2), autodiff::wrt(xi, xj));
101 for (auto k = 0u; k < ny; ++k) {
102 J(k, i) = static_cast<double>(autodiff::detail::derivative<1>(u[k]));
103 H(j, k * nx + i) = autodiff::detail::derivative<2>(u[k]);
104 }
105 });
106 });
107
108 return std::make_tuple(std::move(F), std::move(J), std::move(H));
109 }
110}
