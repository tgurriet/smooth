19template<typename G>
20concept NativeLieGroup = requires
21{
22 // clang-format off
23 typename G::Scalar;
24 typename G::Tangent;
25 typename G::PlainObject;
26 {G::Dof}->std::convertible_to<int>;
27 {G::IsCommutative}->std::convertible_to<bool>;
28} &&
29(!(G::Dof > 0) || requires {
30 {G::Identity()}->std::convertible_to<typename G::PlainObject>;
31 {G::Random()}->std::convertible_to<typename G::PlainObject>;
32}) &&
33(!(G::Dof == -1) || requires (Eigen::Index dof) {
34 {G::Identity(dof)}->std::convertible_to<typename G::PlainObject>;
35 {G::Random(dof)}->std::convertible_to<typename G::PlainObject>;
36}) &&
37(G::Tangent::SizeAtCompileTime == G::Dof) &&
38requires(const G & g1, const G & g2, typename G::Scalar eps) {
39 {g1.dof()}->std::convertible_to<Eigen::Index>; // degrees of freedom at runtime
40 {g1.Ad()}->std::convertible_to<Eigen::Matrix<typename G::Scalar, G::Dof, G::Dof>>;
41 {g1 * g2}->std::convertible_to<typename G::PlainObject>;
42 {g1.inverse()}->std::convertible_to<typename G::PlainObject>;
43 {g1.isApprox(g2, eps)}->std::convertible_to<bool>;
44 {g1.log()}->std::convertible_to<Eigen::Vector<typename G::Scalar, G::Dof>>;
45} &&
46requires(const Eigen::Vector<typename G::Scalar, G::Dof> & a) {
47 {G::ad(a)}->std::convertible_to<Eigen::Matrix<typename G::Scalar, G::Dof, G::Dof>>;
48 {G::exp(a)}->std::convertible_to<typename G::PlainObject>;
49 {G::dr_exp(a)}->std::convertible_to<Eigen::Matrix<typename G::Scalar, G::Dof, G::Dof>>;
50 {G::dr_expinv(a)}->std::convertible_to<Eigen::Matrix<typename G::Scalar, G::Dof, G::Dof>>;
51 {G::d2r_exp(a)}->std::convertible_to<Eigen::Matrix<typename G::Scalar, G::Dof, G::Dof * G::Dof>>;
52 {G::d2r_expinv(a)}->std::convertible_to<Eigen::Matrix<typename G::Scalar, G::Dof, G::Dof * G::Dof>>;
53 // clang-format on
54};
